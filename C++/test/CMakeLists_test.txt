# 测试程序CMakeLists.txt
# 用于编译test_question_system.cpp

cmake_minimum_required(VERSION 3.10)
project(question_system_test CXX)

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找依赖库
find_package(PkgConfig REQUIRED)
pkg_check_modules(CURL REQUIRED libcurl)
pkg_check_modules(JSONCPP REQUIRED jsoncpp)

# 包含头文件目录
include_directories(${CURL_INCLUDE_DIRS})
include_directories(${JSONCPP_INCLUDE_DIRS})

# 添加可执行文件
add_executable(test_question_system ../test_question_system.cpp)

# 链接库
target_link_libraries(test_question_system 
    ${CURL_LIBRARIES} 
    ${JSONCPP_LIBRARIES}
)

# 设置编译选项
target_compile_options(test_question_system PRIVATE ${CURL_CFLAGS_OTHER})
target_compile_options(test_question_system PRIVATE ${JSONCPP_CFLAGS_OTHER})

# 设置链接目录
target_link_directories(test_question_system PRIVATE ${CURL_LIBRARY_DIRS})
target_link_directories(test_question_system PRIVATE ${JSONCPP_LIBRARY_DIRS})
